#!/bin/bash

input_file='/home/zman/RedFolder/TinkerLab/Payloads/Adversary Attack Payloads/Injection/SQL/Generic Error Based .txt'
# Define variables from the provided coordinates
initial_upload_x=504
initial_upload_y=968

browse_x=1276
browse_y=332

choose_file_x=1533
choose_file_y=761

doc_type_selection_x=1599
doc_type_selection_y=382

personal_doc_x=1162
personal_doc_y=456

doc_name_x=1192
doc_name_y=430

notice_to_hr_x=1098
notice_to_hr_y=468

key_1_x=1068
key_1_y=500

upload_x=1592
upload_y=966

# Input file containing the strings to process

# Loop through each string in the file
while IFS= read -r line; do
    # Initial Upload
    xdotool mousemove "$initial_upload_x" "$initial_upload_y" click 1
    sleep 10

    # Browse
    xdotool mousemove "$browse_x" "$browse_y" click 1
    sleep 3

    # Choose File
    xdotool mousemove "$choose_file_x" "$choose_file_y" click 1
    sleep 2

    # Doc Type Selection
    xdotool mousemove "$doc_type_selection_x" "$doc_type_selection_y" click 1
    sleep 3

    # Personal Document
    xdotool mousemove "$personal_doc_x" "$personal_doc_y" click 1   
    sleep 2

    # Document Name
    xdotool mousemove "$doc_name_x" "$doc_name_y" click 1
    xdotool type "$line"
    sleep 1

    # Notice To HR
    xdotool mousemove "$notice_to_hr_x" "$notice_to_hr_y" click 1
    xdotool type "$line"
    sleep 1

    # Key 1
    xdotool mousemove "$key_1_x" "$key_1_y" click 1
    xdotool type "$line"
    sleep 1

    # Upload
    xdotool mousemove "$upload_x" "$upload_y" click 1
    sleep 5  

done < "$input_file"
